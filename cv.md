function disemvowel(str) {
  let result = '';
  const vowels = new Set(['a', 'e', 'i', 'o', 'u']);
  
  for (const char of str) {
    if (!vowels.has(char.toLowerCase())) {
      result += char;
    }
  }
  
  return result;
}
